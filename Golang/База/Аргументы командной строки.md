```go
package main

import (
    "fmt"
    "os"
)

func main() {

    argsWithProg := os.Args
    argsWithoutProg := os.Args[1:]

    arg := os.Args[3]

    fmt.Println(argsWithProg)
    fmt.Println(argsWithoutProg)
    fmt.Println(arg)
}
```
**Терминал:**
```
./main a b c d
```
**Вывод:**
```
[./main a b c d]       
[a b c d]
```
